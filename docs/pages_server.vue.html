            const index = this.currentOrder.findIndex(obj => obj === item)
            if (index !== -1) {
                this.currentOrder.splice(index,1);
            }
            console.log(this.currentOrder)
            localStorage.setItem('currentOrder', JSON.stringify(this.currentOrder));
